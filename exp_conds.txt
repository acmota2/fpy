condition   : eval ':' exp

eval        : eval AND relat
            | eval OR relat
            | NOT eval
            | relat

relat       : ID
            | relat EQ singular
            | relat DIF singular
            | relat '>' singular
            | relat '<' singular
            | relat LTE singular
            | relat GTE singular
            | '(' eval ')'

exp         : exp '+' factor
            | exp '-' factor

factor      : factor '*' term
            | factor '/' term
            | factor INTDIV term

term        : term POW numerical
            | term '%' numerical
            | '(' exp ')'

singular    : STRING
            | rlist
            | rtuple
            | NUM
